draft = false
+++

```sh: ~/.config/lf/lfrc
map gs :select_tagged_files
cmd select_tagged_files &{{
  tagged_files=~/.local/share/lf/tags
  while read line
    do
      tagged_file=${line/%??/}
      if [ $(dirname ${tagged_file}) = ${PWD} ]; then
        lf -remote "send $id toggle '${tagged_file}'"
      fi
    done < ${tagged_files}
}}
```
